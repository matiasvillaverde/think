def is_prime(n):
    """Check if a number is prime."""
    if n < 2:
        return False
    if n == 2:
        return True
    if n % 2 == 0:
        return False
    
    # Check odd divisors up to sqrt(n)
    for i in range(3, int(n**0.5) + 1, 2):
        if n % i == 0:
            return False
    return True

# Find all prime numbers up to 30
primes_up_to_30 = [num for num in range(2, 31) if is_prime(num)]
print(f"Prime numbers up to 30: {primes_up_to_30}")
print(f"Total count: {len(primes_up_to_30)}")

# Verify with a few examples
